## Sprint 7A：enum 穷尽性检查

### 7A.1 symtab API（`compiler/src/symtab.h`）

新增：

```c
// 取 enum 类型的所有 variant sym，按定义顺序返回；返回数量
int sym_enum_variants(SymTable *global_scope, const char *enum_name,
                      Sym **out_variants, int max_variants);
```

实现（`compiler/src/symtab.c`）：

```c
int sym_enum_variants(SymTable *t, const char *enum_name,
                      Sym **out, int max) {
    int n = 0;
    for (size_t i = 0; i < t->nentries && n < max; i++) {
        Sym *s = symtab_load_entry(t, i);
        if (s && s->kind == SYM_VARIANT && s->module) {
            // SYM_VARIANT 的 module 字段在 enum 定义时设为 enum sym 名
            if (strcmp(s->module, enum_name) == 0) {
                out[n++] = s;
            }
        }
    }
    return n;
}
```

**注意**：当前 `SYM_VARIANT` 的 sym 是 `symtab_insert(p->global_scope, name, SYM_VARIANT, type, false, 0)`，**没有 module 字段记录所属 enum**。v0.7 需在 enum 定义时（`parse_type_decl` 的 enum 分支）设 `(*variant_sym).module = arena_strdup(arena, enum_name_str)`。这是**唯一需要回头改的旧代码**——影响 sprint 2 commit 2 翻译的 ast.jhyy 类型 enum 定义（TYPE_KIND→NodeKind 等），需要在 sprint 7A 同步改 jhyy 端 enum 定义填 module 字段。

### 7A.2 sema：穷尽性检查（`compiler/src/sema.c` `case NODE_MATCH`）

修改 line 727 附近的 `case NODE_MATCH`：

```c
case NODE_MATCH: {
    NodeMatch *d = node_match_data(n);
    Type *match_type = infer_type(ctx, d->expr);
    Type *common_type = NULL;

    /* 7A: 穷尽性检查 */
    int covered[(d->narms > 64 ? 64 : d->narms)];  /* 简化：variant 数 < arms 数 */
    int n_covered = 0;
    int wildcard_seen = 0;
    
    if (match_type && match_type->kind == KIND_ENUM && match_type->name) {
        Sym *variants[32];  /* enum 最多 32 variant */
        int nv = sym_enum_variants(ctx->global_scope, match_type->name,
                                   variants, 32);
        /* 跟踪每个 variant 是否被 arm 覆盖 */
        int8_t seen[32] = {0};
        
        for (size_t i = 0; i < d->narms; i++) {
            NodeMatchArm *arm = node_match_arm_data(d->arms[i]);
            Node *pat = arm->pattern;
            
            if (pat->kind == NODE_PATTERN_WILDCARD) {
                wildcard_seen = 1;
            } else if (pat->kind == NODE_PATTERN_ENUM_VARIANT) {
                NodePatternEnum *pe = node_pattern_enum_data(pat);
                const char *vname = pe->variant_name;
                /* 短名或 Enum::Variant 都行，提取末尾 ident */
                if (strstr(vname, "::")) vname = strstr(vname, "::") + 2;
                for (int j = 0; j < nv; j++) {
                    if (strcmp(variants[j]->name, vname) == 0) {
                        if (seen[j]) {
                            sema_error(ctx, pat->loc,
                                "duplicate match arm for variant '%s'", vname);
                        }
                        seen[j] = 1;
                        break;
                    }
                }
            }
            /* NODE_PATTERN_IDENT 也算覆盖（绑定到变量） */
            
            Type *body_type = infer_type(ctx, arm->body);
            if (i == 0) common_type = body_type;
            else if (!type_eq(common_type, body_type) &&
                     common_type->kind != KIND_VOID && body_type->kind != KIND_VOID)
                sema_error(ctx, n->loc, "match arm type mismatch");
        }
        
        /* 检查未覆盖 variant */
        if (!wildcard_seen) {
            for (int j = 0; j < nv; j++) {
                if (!seen[j]) {
                    sema_error(ctx, n->loc,
                        "non-exhaustive match: missing variant '%s'",
                        variants[j]->name);
                }
            }
        }
    } else {
        /* 非 enum：现有逻辑不变 */
        for (size_t i = 0; i < d->narms; i++) {
            NodeMatchArm *arm = node_match_arm_data(d->arms[i]);
            Type *body_type = infer_type(ctx, arm->body);
            if (i == 0) common_type = body_type;
            else if (!type_eq(common_type, body_type) &&
                     common_type->kind != KIND_VOID && body_type->kind != KIND_VOID)
                sema_error(ctx, n->loc, "match arm type mismatch");
        }
    }
    
    n->type = common_type ? common_type : type_void();
    return n->type;
}
```

### 7A.3 parser：短名 variant pattern（`compiler/src/parser.c` `parse_pattern`）

修改 `TOKEN_IDENT` 分支：

```c
case TOKEN_IDENT: {
    Token name = consume(p, "identifier");
    const char *vname = tok_name(p, name);
    
    /* 7A: 短名 variant pattern：先查 SYM_VARIANT */
    Sym *sym = symtab_lookup_local(p->global_scope, vname);
    if (sym && sym->kind == SYM_VARIANT) {
        /* 是 enum variant 短名 */
        return ast_new_pattern_enum_variant(p->arena, name.loc, sym, NULL);

    sym = symtab_lookup(p->current_scope, vname);

    Node *lo = ast_new_pattern_ident(p->arena, name.loc, sym);
    return lo;
}
```

```c
static Node *parse_const_decl(Parser *p) {
    SourceLoc loc = peek(p).loc;
    advance(p);  /* consume 'const' */
    
    Token name = expect(p, TOKEN_IDENT, "const name");
    const char *cname = tok_name(p, name);
    
    expect(p, TOKEN_COLON, ":");
    Node *type_annot = parse_type(p);  /* 必须是 [T; N] */
    
    expect(p, TOKEN_EQ, "=");
    Node *init = parse_array_literal(p);  /* 复用现有 array lit parse */
    
    expect(p, TOKEN_SEMICOLON, ";");
    
    Sym *sym = symtab_insert(p->global_scope, cname, SYM_CONST, NULL, false, 0);
    return ast_new_const_decl(p->arena, loc, sym, type_annot, init);
}
```

### 7B.4 sema

```c
case NODE_CONST_DECL: {
    NodeConstDecl *d = node_const_decl_data(n);
    Type *decl_type = resolve_type_node(ctx, d->type);
    d->sym->type = decl_type;
    
    /* 校验 [T; N] */
    if (decl_type->kind != KIND_ARRAY) {
        sema_error(ctx, n->loc, "const declaration must be an array type");
        return type_void();
    }
    Type *elem_type = decl_type->array.elem;
    int n_elems = decl_type->array.count;  /* 必须是字面量 */
    
    /* 校验 elem 类型限定 */
    if (elem_type->kind == KIND_POINTER || elem_type->kind == KIND_ENUM) {
        sema_error(ctx, n->loc,
            "const array element type %s not supported (only primitives + structs)",
            type_to_string(elem_type));
    }
    
    /* 校验 init 是 NODE_ARRAY_LIT */
    if (d->init->kind != NODE_ARRAY_LIT) {
        sema_error(ctx, n->loc, "const declaration requires array literal");
    }
    NodeArrayLit *arr = node_array_lit_data(d->init);
    if ((int)arr->nelems != n_elems) {
        sema_error(ctx, n->loc,
            "const array length mismatch: declared [%d], got %zu",
            n_elems, arr->nelems);
    }
    
    /* 校验每个 elem 是 const expression */
    for (size_t i = 0; i < arr->nelems; i++) {
        Node *e = arr->elems[i];
        if (!is_const_expr(e, ctx)) {
            sema_error(ctx, e->loc,
                "const array element must be a const expression");
        }
    }
    
    /* 存到 ctx->global_consts 列表（codegen 用） */
    ctx->global_consts[ctx->n_global_consts++] = d;
    
    n->type = type_void();
    return n->type;
}

static int is_const_expr(Node *n, SemaContext *ctx) {
    switch (n->kind) {
        case NODE_INT: case NODE_FLOAT: case NODE_BOOL:
        case NODE_STRING: return 1;
        case NODE_STRUCT_LIT: /* 每个字段递归 */
            NodeStructLit *sl = node_struct_lit_data(n);
            for (size_t i = 0; i < sl->nfields; i++) {
                if (!is_const_expr(sl->fields[i].value, ctx)) return 0;
            }
            return 1;
        case NODE_IDENT: {
            /* 引用其他 SYM_CONST */
            Sym *s = node_ident_data(n)->sym;
            return s && s->kind == SYM_CONST;
        }
        default: return 0;
    }
}
```

### 7B.5 codegen：emit data section

`compiler/src/codegen.c` `codegen_module` 顶层循环：先扫一遍所有 NODE_CONST_DECL，emit data section，再 emit functions。

```c
case NODE_CONST_DECL: {
    NodeConstDecl *d = node_const_decl_data(n);
    Type *arr_type = d->sym->type;
    Type *elem_type = arr_type->array.elem;
    int n = arr_type->array.count;
    size_t elem_size = type_size(elem_type);
    char qbe_type_char;
    switch (elem_size) {
        case 1: qbe_type_char = 'b'; break;
        case 2: qbe_type_char = 'h'; break;
        case 4: qbe_type_char = 'w'; break;
        case 8: qbe_type_char = 'l'; break;
        default: qbe_type_char = (elem_size >= 8 ? 'l' : 'w'); break;
    }
    
    ir_emit(cg->ir, "data $%s = { ", d->sym->name);
    NodeArrayLit *arr = node_array_lit_data(d->init);
    for (int i = 0; i < n; i++) {
        Node *e = arr->elems[i];
        if (elem_type->kind == KIND_STRUCT) {
            /* struct 字面量：递归 emit fields */
            ir_emit_struct_lit_data(cg->ir, e, elem_type);
        } else if (elem_type->kind == KIND_PRIMITIVE) {
            ir_emit_prim_data(cg->ir, e, elem_type);
        }
        if (i < n - 1) ir_emit(cg->ir, ", ");
    }
    ir_emit(cg->ir, " }\n");
    break;
}
```

## Sprint 7C：文档同步

### 7C.1 lang-spec

`docs/abis/jhyy-lang-spec-v1.0.0.md` → v1.1.0（bump minor version 因为新增语法）：

**§ 11.3 match** 加：

> **穷尽性检查（v0.7 起）**：match 表达式类型是 enum 时，**必须覆盖所有 variant**。漏一个 variant → 编译错误 `non-exhaustive match: missing variant 'X'`。`_` 通配符可作兜底，但必须放在最后且仅一次。
>
> 例：
> ```rust
> type Option = enum { Some(i32), None };
> let opt = Option::Some(5);
> let v = match opt {
>     Option::Some(x) => x,
>     Option::None => 0,
> };  /* ✅ 全覆盖 */
>
> match opt {
>     Option::Some(x) => x,  /* ❌ 漏 None，编译错误 */
> };
> ```
>
> **短名 variant pattern**：`match opt { Some(x) => x, None => 0 }` 等价于 `Option::Some(x) => x, Option::None => 0`——编译器自动在 SYM_VARIANT 查短名。

**新增 § 13 顶层 const 声明**：

> ### 13.1 const 数组
>
> ```rust
> const NAME: [T; N] = [expr, expr, ...];
> ```
>
> - T 限定：**原语 (i32/i64/u8/...) + struct (字段必须也是原语或 struct)**——不支持 enum / 指针 / slice
> - N 必须是字面量整数常量
> - 初值必须是 const 表达式（字面量 / struct 字面量 / 其他 const 引用）
> - codegen emit 到 QBE `data` 段，运行时零成本加载
>
> 例：
> ```rust
> const ASCII_LOWER: [u8; 26] = [97, 98, ..., 122];
>
> type RGB = struct { r: i32, g: i32, b: i32 };
> const PALETTE: [RGB; 3] = [
>     RGB { r: 1, g: 2, b: 3 },
>     RGB { r: 4, g: 5, b: 6 },
>     RGB { r: 7, g: 8, b: 9 },
> ];
>
> fn main_jhyy() -> i32 {
>     ASCII_LOWER[25] as i32 + PALETTE[2].b  /* 122 + 9 = 131 */
> }
> ```

## 风险 + 缓解

| 风险 | 影响 | 缓解 |
|------|------|------|
| 7A 短名 variant pattern 跟局部变量冲突 | 编译器歧义 | sema 优先查 SYM_VARIANT；如果 SYM_VARIANT 和 SYM_VAR 同名（不应该出现但防御）报错 |
| 7A jhyy 端 enum 定义没设 module 字段（sprint 2 commit 2 翻译时漏了） | enum 穷尽性检查拿不到 variant 列表 | 7A 实施时同步回头改 ast.jhyy enum 定义加 module 字段 |
| 7B const 数组的 struct 字段是 *u8（name） | 不支持，编译错误 | v0.7 限定 struct 字段为原语；name 字段走 `let name = "...";` 然后引用（SYM_VAR 不能在 const init）—— v0.8+ 加 const string 支持 |
| 7B QBE data 段 64KB 限制 | 大表 emit 失败 | v0.7 不考虑；超过 1KB 的表走 v0.8+ 拆分 |
| 7B const 初值引用其他 const | 顺序敏感（`const A = [...]; const B = [..., A[0]];` A 必须先定义） | parser 单 pass 顺序处理（jhyy 已有顺序敏感，OK） |
| 7A + 7B 都改 sema.c 同一文件 | merge 冲突 | 7A 在前、7B 在后，线性无冲突；commit 顺序：7A → 7B → 7C |
